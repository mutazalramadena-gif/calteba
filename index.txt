<script>
function calc(){
  const v = parseFloat(document.getElementById('invoice').value || 0);
  const c = parseInt(document.getElementById('count').value || 1);
  const r = parseFloat(document.getElementById('rate').value || 0.006);

  // حساب دقيق وعرض الناتج فقط
  const total = (v * r) + (0.50 * c);

  document.getElementById('output').style.display='block';
  document.getElementById('output').innerText = total.toFixed(5);
}
</script>
